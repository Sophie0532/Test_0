Private Sub CategoriaListBox_Click()
    If CategoriaListBox.ListIndex > -1 Then
        ' Guarda el texto visible como nombre de la categoría
        GLOBAL_CATEGORY = CategoriaListBox.Value
        
        ' Guarda el código de categoría, si viene desde la consulta
        ' Supone que athena_category_list tiene la forma: (CAT_COD, CAT)
        If Not IsEmpty(athena_category_list) Then
            GLOBAL_CATEGORY_COD = athena_category_list(CategoriaListBox.ListIndex)(0)
        End If
    End If
End Sub
